EXERCISE 9(a)
AIM:

Adding two large integers which are represented in linked list fashion.

PROGRAM:
#include<stdio.h>
#include<stdlib.h>
#include<conio.h>
#include<ctype.h>

struct node{
    int data;
    struct node*next;
};

void insert(struct node**p,int num){
    struct node*temp;
    if(*p==NULL){
        (*p)=(struct node*)malloc(sizeof(struct node));
        (*p)->next=NULL;
        (*p)->data=num;
      }
    else{
        temp=(struct node*)malloc(sizeof(struct node));
        temp->next=(*p);
        (*p)=temp;
        (*p)->data=num;
     }
 }
